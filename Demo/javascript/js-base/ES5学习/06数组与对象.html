本章主要讲的是数组和对象的综合使用

一、数组里面放对象

var str =[{name:"好兄弟",ages:30}];

document.write(str[0].name);

好兄弟

二、对象里面放数组

var obj = {type:["童鞋","拖鞋","皮鞋","运动鞋"],2:["pink","green","yellow","red"]};

document.write(obj.type);获取对象

document.write(obj[2]);如果对象为数字的话必须这么写

童鞋,拖鞋,皮鞋,运动鞋
pink,green,yellow,red

三、利用for循环把数组里面的值取出来

var obj = {atype:["童鞋","拖鞋","皮鞋","运动鞋"],2:["pink","green","yellow","red"]};

for(var i=0; i<obj.atype.length; i++)

{

var a1 = obj.atype[0];

var a2 = obj.atype[1];

var a3 = obj.atype[2];

var a4 = obj.atype[3];

}

document.write("第一个值:"+a1+"---第二个值:"+a2+"---第三个值:"+a3+"---第四个值:"+a4);


四、对数组内容进行分隔

var obj = {atype:["101#童鞋","102#拖鞋"]};

var a1 = obj.atype[0];

document.write(a1);

var fen = obj.atype[0].split("#");

ne1 = fen[0];

ne2 = fen[1];

document.write(ne1+"-----"+ne2);

101#童鞋
101-----童鞋
