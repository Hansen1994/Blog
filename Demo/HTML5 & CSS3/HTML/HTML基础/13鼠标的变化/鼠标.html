鼠标的变化

浏览器默认值
cursor:auto;

等待/沙漏
cursor:wait;

帮助
cursor:help;

小手
cursor: pointer;

十字线
cursor:crosshair;

无法解释
cursor:no-drop;

文字/编辑
cursor:text;

可移动对象
cursor:move;

向上可改变大小
cursor:n-resize;

向下可改变大小
cursor:s-resize;

向右可改变大小
cursor:e-resize;

向左可改变的小
cursor:w-resize;

向上右可改变大小
cursor:ne-resize;

向上左可改变大小
cursor:nw-resize;

向下右可改变的小
cursor:se-resize;

向下左可改变的小
cursor:sw-resize;

禁止
cursor:not-allowed;

处理中
cursor:progress;

系统默认
cursor:default;

用户自定义
cursor:url('#');
